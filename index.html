// ============================================================ // FUNDUSCOPY TCA SYSTEM — Google Apps Script (V5) // Dengan pagination untuk elak freeze // ============================================================ const SOURCE_SHEET = "ACTIVE PT NDR "; const TARGET_SHEET = "FUNDUSCOPY TCA"; const COL_NAME = 1; const COL_IC = 2; const COL_STATUS = 3; const COL_REMARKS = 11; const COLUMNS = [ "ID","Nama Pesakit","No. IC","No. Telefon","Jantina","Umur","MRN", "Tarikh TCA","Masa","Klinik","Keputusan Fundus","Mata Diperiksa", "VA OD","VA OS","BP","HbA1c","Tindakan / Outcome","Tarikh TCA Seterusnya", "Doktor","Hospital Rujukan","Tarikh Rujukan","No. Rujukan", "Sebab Enggan","Mata Buta","Sebab Buta","Nota","Status NDR","Timestamp" ]; function getSourceSheet() { return SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SOURCE_SHEET); } function getTargetSheet() { return SpreadsheetApp.getActiveSpreadsheet().getSheetByName(TARGET_SHEET); } function ensureHeaders() { const sheet = getTargetSheet(); if (!sheet) return; const firstRow = sheet.getRange(1, 1, 1, COLUMNS.length).getValues()[0]; if (firstRow.every(c => c === "")) { sheet.getRange(1, 1, 1, COLUMNS.length).setValues([COLUMNS]); sheet.getRange(1, 1, 1, COLUMNS.length) .setBackground("#0b5394").setFontColor("#ffffff").setFontWeight("bold"); sheet.setFrozenRows(1); } } function makeResponse(data) { const output = ContentService.createTextOutput(JSON.stringify(data)); output.setMimeType(ContentService.MimeType.JSON); return output; } function mapRemarksToOutcome(remarks) { if (!remarks) return ""; const r = remarks.toString().toUpperCase().trim(); if (r.includes("REFER HTI") || r.includes("REFER EYE")) return "Rujukan Eye HTI"; if (r.includes("HTI F/UP") || r.includes("HTI FU") || r.includes("UNDER HTI") || r.includes("HTI FOLLOW")) return "Eye Under HTI Follow-Up"; if (r.includes("GP F/UP") || r.includes("UNDER GP")) return "Eye Under GP Follow-Up"; if (r.includes("BLIND") || r.includes("BUTA")) return "Pesakit Buta"; return ""; } function nearestFundusDay(date) { const d = new Date(date); const day = d.getDay(); if (day === 0) d.setDate(d.getDate() + 1); else if (day === 5) d.setDate(d.getDate() - 1); else if (day === 6) d.setDate(d.getDate() - 2); return d; } function calcNextTCA(outcome) { const today = new Date(); let target = null; if (outcome === "3 Bulan") { target = new Date(today); target.setMonth(target.getMonth() + 3); } else if (outcome === "6 Bulan") { target = new Date(today); target.setMonth(target.getMonth() + 6); } else if (outcome === "1 Tahun") { target = new Date(today); target.setFullYear(target.getFullYear() + 1); } if (!target) return ""; return Utilities.formatDate(nearestFundusDay(target), Session.getScriptTimeZone(), "yyyy-MM-dd"); } // ── GET with pagination ────────────────────────────────────── function doGet(e) { try { const action = e.parameter.action; const page = parseInt(e.parameter.page || "0"); const limit = parseInt(e.parameter.limit || "100"); if (action === "import") return makeResponse(importFromActiveNDR()); if (action === "count") { const sheet = getTargetSheet(); if (!sheet) return makeResponse({ count: 0 }); const count = Math.max(0, sheet.getLastRow() - 1); return makeResponse({ count: count }); } ensureHeaders(); const sheet = getTargetSheet(); if (!sheet) return makeResponse({ error: "Sheet tidak dijumpai: " + TARGET_SHEET }); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return makeResponse({ rows: [], total: 0, page: 0 }); const allRows = data.slice(1).map(row => { const obj = {}; COLUMNS.forEach((col, i) => { obj[col] = row[i] !== undefined ? String(row[i]) : ""; }); return obj; }); const total = allRows.length; // If page=-1, return ALL (for search/filter) if (page === -1) return makeResponse({ rows: allRows, total: total }); // Return paginated const start = page * limit; const pageRows = allRows.slice(start, start + limit); return makeResponse({ rows: pageRows, total: total, page: page, limit: limit, hasMore: start + limit < total }); } catch (err) { return makeResponse({ error: err.message }); } } // ── POST ───────────────────────────────────────────────────── function doPost(e) { try { ensureHeaders(); const payload = JSON.parse(e.postData.contents); const action = payload.action; if (action === "insert") return makeResponse(insertRow(payload.row)); if (action === "update") return makeResponse(updateRow(payload.id, payload.row)); if (action === "delete") return makeResponse(deleteRow(payload.id)); return makeResponse({ error: "Unknown action: " + action }); } catch (err) { return makeResponse({ error: err.message }); } } function insertRow(row) { const sheet = getTargetSheet(); const values = COLUMNS.map(col => { if (col === "Timestamp") return new Date().toLocaleString("ms-MY"); return row[col] !== undefined ? row[col] : ""; }); sheet.appendRow(values); return { success: true, id: row["ID"] }; } function updateRow(id, row) { const sheet = getTargetSheet(); const data = sheet.getDataRange().getValues(); const idCol = COLUMNS.indexOf("ID"); for (let i = 1; i < data.length; i++) { if (String(data[i][idCol]) === String(id)) { const values = COLUMNS.map((col, idx) => { if (col === "Timestamp") return new Date().toLocaleString("ms-MY"); return row[col] !== undefined ? row[col] : data[i][idx]; }); sheet.getRange(i + 1, 1, 1, COLUMNS.length).setValues([values]); return { success: true }; } } return { success: false, error: "Row tidak dijumpai" }; } function deleteRow(id) { const sheet = getTargetSheet(); const data = sheet.getDataRange().getValues(); const idCol = COLUMNS.indexOf("ID"); for (let i = data.length - 1; i >= 1; i--) { if (String(data[i][idCol]) === String(id)) { sheet.deleteRow(i + 1); return { success: true }; } } return { success: false, error: "Row tidak dijumpai" }; } function importFromActiveNDR() { const source = getSourceSheet(); const target = getTargetSheet(); if (!source) return { success: false, error: "Sheet sumber tidak dijumpai: [" + SOURCE_SHEET + "]" }; if (!target) return { success: false, error: "Sheet target tidak dijumpai: [" + TARGET_SHEET + "]" }; ensureHeaders(); const targetData = target.getDataRange().getValues(); const existingICs = new Set(); const icColIdx = COLUMNS.indexOf("No. IC"); for (let i = 1; i < targetData.length; i++) { if (targetData[i][icColIdx]) existingICs.add(String(targetData[i][icColIdx]).trim()); } const sourceData = source.getDataRange().getValues(); let imported = 0; let skipped = 0; for (let i = 2; i < sourceData.length; i++) { const row = sourceData[i]; const name = String(row[COL_NAME] || "").trim(); const ic = String(row[COL_IC] || "").trim(); if (!name || !ic) { skipped++; continue; } if (existingICs.has(ic)) { skipped++; continue; } const remarks = String(row[COL_REMARKS] || "").trim(); const status = String(row[COL_STATUS] || "").trim(); const outcome = mapRemarksToOutcome(remarks); const nextTCA = calcNextTCA(outcome); let age = "", gender = ""; const icClean = ic.replace(/-/g, ""); if (icClean.length >= 6) { const yy = parseInt(icClean.slice(0, 2)); const mm = parseInt(icClean.slice(2, 4)); const dd = parseInt(icClean.slice(4, 6)); const fullYear = yy > (new Date().getFullYear() % 100) ? 1900 + yy : 2000 + yy; const dob = new Date(fullYear, mm - 1, dd); let calcAge = new Date().getFullYear() - dob.getFullYear(); const mDiff = new Date().getMonth() - dob.getMonth(); if (mDiff < 0 || (mDiff === 0 && new Date().getDate() < dob.getDate())) calcAge--; if (calcAge >= 0 && calcAge <= 120) age = calcAge; } if (icClean.length >= 12) { const lastDigit = parseInt(icClean.slice(11, 12)); if (!isNaN(lastDigit)) gender = lastDigit % 2 === 1 ? "Lelaki" : "Perempuan"; } const uid = Date.now().toString(36) + Math.random().toString(36).slice(2, 5); const newRow = COLUMNS.map(col => { if (col === "ID") return uid; if (col === "Nama Pesakit") return name; if (col === "No. IC") return ic; if (col === "Umur") return age; if (col === "Jantina") return gender; if (col === "Tindakan / Outcome") return outcome; if (col === "Tarikh TCA Seterusnya") return nextTCA; if (col === "Nota") return remarks; if (col === "Status NDR") return status; if (col === "Timestamp") return new Date().toLocaleString("ms-MY"); return ""; }); target.appendRow(newRow); existingICs.add(ic); imported++; Utilities.sleep(30); } return { success: true, imported: imported, skipped: skipped, message: imported + " pesakit berjaya diimport!" }; } function debugSheets() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheets = ss.getSheets(); Logger.log("Semua tab: " + sheets.map(s => "'" + s.getName() + "'").join(", ")); }
